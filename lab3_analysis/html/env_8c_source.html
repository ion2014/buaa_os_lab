311  printf("[%08x] free env %08x\n", curenv ? curenv->env_id : 0, e->env_id);
312 
313  /* Hint: Flush all mapped pages in the user portion of the address space */
314  for (pdeno = 0; pdeno < PDX(UTOP); pdeno++) {
315  /* Hint: only look at mapped page tables. */
316  if (!(e->env_pgdir[pdeno] & PTE_V)) {
317  continue;
318  }
319  /* Hint: find the pa and va of the page table. */
320  pa = PTE_ADDR(e->env_pgdir[pdeno]);
321  pt = (Pte *)KADDR(pa);
322  /* Hint: Unmap all PTEs in this page table. */
323  for (pteno = 0; pteno <= PTX(~0); pteno++)
324  if (pt[pteno] & PTE_V) {
325  page_remove(e->env_pgdir, (pdeno << PDSHIFT) | (pteno << PGSHIFT));
326  }
327  /* Hint: free the page table itself. */
328  e->env_pgdir[pdeno] = 0;
329  page_decref(pa2page(pa));
330  }
331  /* Hint: free the page directory. */
332  pa = e->env_cr3;
333  e->env_pgdir = 0;
334  e->env_cr3 = 0;
335  page_decref(pa2page(pa));
336  /* Hint: return the environment to the free list. */
337  e->env_status = ENV_FREE;
