38 /* Overview:
39  Allocate `n` bytes physical memory with alignment `align`, if `clear` is set, clear the
40  allocated memory.
41  This allocator is used only while setting up virtual memory system.
42 
43  Post-Condition:
44  If we're out of memory, should panic, else return this address of memory we have allocated.*/
45 static void *alloc(u_int n, u_int align, int clear)
46 {
47  extern char end[];
48  u_long alloced_mem;
49 
50  /* Initialize `freemem` if this is the first time. The first virtual address that the
51  * linker did *not* assign to any kernel code or global variables. */
52  if (freemem == 0) {
53  freemem = (u_long)end;//end=0x80400000
54  }
55 
56  /* Step 1: Round up `freemem` up to be aligned properly */
57  freemem = ROUND(freemem, align);
58 
59  /* Step 2: Save current value of `freemem` as allocated chunk. */
60  alloced_mem = freemem;
61 
62  /* Step 3: Increase `freemem` to record allocation. */
63  freemem = freemem + n;
64 
65  /* Step 4: Clear allocated chunk if parameter `clear` is set. */
66  if (clear) {
67  bzero((void *)alloced_mem, n);
68  }
69 
70  // We're out of memory, PANIC !!
71  if (PADDR(freemem) >= maxpa) {
72  panic("out of memorty\n");
73  return (void *)-E_NO_MEM;
74  }
75 
76  /* Step 5: return allocated chunk. */
77  return (void *)alloced_mem;
78 }

280  entry to *ppte, and return 0, indicating success.

284  whether this function execute successfully or not.

286 int pgdir_walk(Pde *pgdir, u_long va, int create, Pte **ppte)
287 {

318 /*Overview:
319  Map the physical page 'pp' at virtual address 'va'.
320  The permissions (the low 12 bits) of the page table entry should be set to 'perm|PTE_V'.
321 
322  Post-Condition:
323  Return 0 on success
324  Return -E_NO_MEM, if page table couldn't be allocated
325 
326  Hint:
327  If there is already a page mapped at `va`, call page_remove() to release this mapping.
328  The `pp_ref` should be incremented if the insertion succeeds.*/
329 int page_insert(Pde *pgdir, struct Page *pp, u_long va, u_int perm)
330 {
331  u_int PERM;
332  Pte *pgtable_entry;
333  PERM = perm | PTE_V;
334 
335  /* Step 1: Get corresponding page table entry. */
336  pgdir_walk(pgdir, va, 0, &pgtable_entry);
337 
338  if (pgtable_entry != 0 && (*pgtable_entry & PTE_V) != 0) {
339  if (pa2page(*pgtable_entry) != pp) {
340  page_remove(pgdir, va);
341  } else {
342  tlb_invalidate(pgdir, va);
343  *pgtable_entry = (page2pa(pp) | PERM);
344  return 0;
345  }
346  }
347 
348  /* Step 2: Update TLB. */
349  tlb_invalidate(pgdir, va);
350 
351  /* Step 3: Do check, re-get page table entry to validate the insertion. */
352  if (pgdir_walk(pgdir, va, 1, &pgtable_entry) != 0) {
353  return -E_NO_MEM; // panic ("page insert failed .\n");
354  }
355 
356  *pgtable_entry = (page2pa(pp) | PERM);
357  pp->pp_ref++;
358  return 0;
359 }
